class A(object):
    def foo(self, x):
        print("{}, {}".format(self, x))

    @classmethod
    def class_foo(cls, x):
        print("classmethod: {}, {}".format(cls, x))

    @staticmethod
    def static_foo(x):
        print("staticmethod: {}".format(x))

a = A()

A.static_foo(3)

def runtime(text):
    def decorator(func):
        @functools.wraps(func):
        def wrapper(*args, **kwargs)
            start_time = time.time()
            rest = func(*args, **kwargs)
            end_time = time.time()
            print(text, end_time-start_time)

        return wrapper

    return decorator

# 类装饰器
class runtime(object):
    def __init__(self, text):
        self.text = text

        @functools.wraps(func)

            rest = func(*args, **kwargs)
            end_time = time.time()
            print(self.text, end_time-start_time)

        return wrapper

def singletion(cls):
    instances = {}

    def getInstance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    return getInstance

import time

from threading import Thread


def countdown(name, n):
    while n > 0:
        print('{} countdown: {}'.format(name, n))
        time.sleep(random.random())

        n -= 1


t = Thread(target=countdown, args=('thread 1', 5,))
t2 = Thread(target=countdown, args=('thread 2', 5,))
t.start()
t2.start()

t.join()
t2.join()
print('success')


class CountdownThread(Thread):
    def __init__(self, name, n):
        super(CountdownThread, self).__init__()
        self.name = name
        self.n = n

    def run(self):
        while self.n > 0:
            print('{} countdown: {}'.format(self.name, self.n))
            time.sleep(random.random())

            self.n -= 1


t = CountdownThread('thread 1', 5)
t2 = CountdownThread('thread 2', 5)

def simple_coroutine():

    return_value = 'haha'
    x = yield return_value

def getgeneratorstate(generator):
    if generator.gi_running:
        return 'GEN_RUNNING'
    if generator.gi_frame is None:
        return 'GEN_CLOSED'
    if generator.gi_frame.f_lasti == -1:
        return 'GEN_CREATED'
    return 'GEN_SUSPENDED'
